InfoDb = []

InfoDb.append({
    "FirstName": "Lydia",
    "LastName": "Cho",
    "DOB": "April 12",
    "Residence": "San Diego",
    "Email": "lydiac55288@stu.powayusd.com"
    })
print(InfoDb)

def print_data(d_rec):
    print(d_rec["FirstName"], d_rec["LastName"])  
    print("\t", "Residence:", d_rec["Residence"]) 
    print("\t", "Birth Day:", d_rec["DOB"])
    print()


def for_loop():
    print("For loop output\n")
    for record in InfoDb:
        print_data(record)

for_loop()

def while_loop():
    print("While loop output\n")
    i = 0
    while i < len(InfoDb):
        record = InfoDb[i]
        print_data(record)
        i += 1
    return

while_loop()

def recursive_loop(i):
    if i < len(InfoDb):
        record = InfoDb[i]
        print_data(record)
        recursive_loop(i + 1)
    return
    
print("Recursive loop output\n")
recursive_loop(0)
